```bash
# Extract the new tag from G-05.1-1 evidence (strip leading 'v')

echo "Probing GHCR for tag: $NEW_TAG"

EVIDENCE=.planning/phases/05.1-ci-autotag-chain-repair/evidence/sc-04-ghcr-probe.txt
{
  echo "G-05.1-2 / SC#4 evidence (captured $(date -u +%Y-%m-%dT%H:%M:%SZ))"
  echo "================================================================"
  echo ""
  echo "Target tag: $NEW_TAG (from G-05.1-1 evidence)"
  echo "Image: ghcr.io/tom333/arr-stack-arrconf"
  echo ""
  echo "=== Probe 1: OCI image-index Accept header (canonical, per CLAUDE.md lesson) ==="
  curl -sI \
    -H "Accept: application/vnd.oci.image.index.v1+json" \
    "https://ghcr.io/v2/tom333/arr-stack-arrconf/manifests/${NEW_TAG}" \
    | head -5
  echo ""
  echo "=== Probe 2: Anonymous Docker manifest v2 (fallback verification) ==="
  curl -sI \
    -H "Accept: application/vnd.docker.distribution.manifest.v2+json" \
    "https://ghcr.io/v2/tom333/arr-stack-arrconf/manifests/${NEW_TAG}" \
    | head -5
  echo ""
  echo "=== Probe 3: Confirm v0.3.0 is still an orphan (D-05.1-ORPHAN-01) ==="

    -H "Accept: application/vnd.oci.image.index.v1+json" \

    | head -5
} | tee "$EVIDENCE"
```
