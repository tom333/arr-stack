```python
class SeerrInstance(BaseModel):
    base_url: str
    sonarr_service: SeerrSonarrServiceSection  # required

class SeerrSonarrServiceSection(BaseModel):
    hostname: str
    port: int = 8989
    useSsl: bool = False
    activeProfileId: int                    # required (integer ID — from Sonarr quality profiles)
    activeDirectory: str                    # required
    activeAnimeProfileId: int | None = None # optional — integer ID of Sonarr "Anime" profile
    activeAnimeDirectory: str | None = None # optional
    tags: list[int] = []                    # Sonarr tag IDs for non-anime requests
    animeTags: list[int] = []               # Sonarr tag IDs for anime requests
    is4k: bool = False
    isDefault: bool = True
    enableSeasonFolders: bool = False
    externalUrl: str = ""
    syncEnabled: bool = True
    preventSearch: bool = False
    tagRequests: bool = True
```
