Create `.github/workflows/chart-lint.yml` with the full body below. The workflow has 2 jobs: `lint` (PR + push) and `tag` (push-to-main only, depends on lint).

```yaml
name: chart-lint

on:
  push:
    branches: [main]
    paths:
      - "charts/**"
      - "examples/values-prod.yaml"
      - ".github/workflows/chart-lint.yml"
      - "renovate.json"
      - "tools/scripts/**"
  pull_request:
    paths:
      - "charts/**"
      - "examples/values-prod.yaml"
      - ".github/workflows/chart-lint.yml"
      - "renovate.json"
      - "tools/scripts/**"

jobs:
  lint:
    runs-on: ubuntu-24.04
    permissions:
      contents: read
    steps:
      - uses: actions/checkout@v4

      - name: Set up Helm
        uses: azure/setup-helm@v4
        with:
          version: v3.18.0  # >=3.18 required by app-template 5.0.0

      - name: Helm dependency update
        run: |
          helm dependency update charts/arr-stack/

      - name: Helm lint (with values.schema.json validation)
        run: |
          helm lint charts/arr-stack/ -f examples/values-prod.yaml

      - name: Helm template
        run: |
          helm template arr-stack charts/arr-stack/ \
            -f examples/values-prod.yaml \
            > /tmp/manifests.yaml

      - name: Install kubeconform
        run: |
          curl -sSL https://github.com/yannh/kubeconform/releases/latest/download/kubeconform-linux-amd64.tar.gz \
            | sudo tar xz -C /usr/local/bin/

      - name: kubeconform
        run: |
          kubeconform \
            -kubernetes-version 1.33.0 \
            -strict \
            -ignore-missing-schemas \
            /tmp/manifests.yaml

      - name: Renovate annotation contract (every repository: line has a renovate annotation above)
        run: |
          chmod +x tools/scripts/check-renovate-annotations.sh
          tools/scripts/check-renovate-annotations.sh charts/arr-stack/values.yaml

      - name: ":latest tag guard (C9 no :latest invariant)"
        run: |
          if grep -q 'tag: latest' charts/arr-stack/values.yaml; then
            echo "ERROR: 'tag: latest' found in values.yaml — pin to a semver per D-04-PIN-01"
            grep -n 'tag: latest' charts/arr-stack/values.yaml
            exit 1
          fi
          echo "OK: no :latest tags in values.yaml"

      - name: "cronJobConfig regression guard (v4-era key forbidden in v5 chart)"
        run: |
          if grep -q 'cronJobConfig:' charts/arr-stack/values.yaml; then
            echo "ERROR: 'cronJobConfig:' is the v4 key — app-template 5.0.0 uses 'cronjob:' (RESEARCH Pitfall 5)"
            grep -n 'cronJobConfig:' charts/arr-stack/values.yaml
            exit 1
          fi
          echo "OK: no v4-era cronJobConfig: keys in values.yaml"

      - name: customManagers regex synthetic test (>= 10 matches in values.yaml)
        run: |
          python3 <<'PY'
          import json, re, sys
          with open('renovate.json') as f:
              cfg = json.load(f)
          managers = [m for m in cfg.get('customManagers', []) if m.get('customType') == 'regex']
          if not managers:
              print('ERROR: no customManagers/regex entry in renovate.json')
              sys.exit(1)
          with open('charts/arr-stack/values.yaml') as f:
              values = f.read()
          total_matches = 0
          for mgr in managers:
              for pat in mgr.get('matchStrings', []):
                  # Renovate uses (?<name>...) named-group syntax; Python supports it too
                  matches = re.findall(pat, values)
                  total_matches += len(matches)
                  print(f'pattern matched {len(matches)} times: {pat!r}')
          if total_matches < 10:
              print(f'ERROR: expected >= 10 image matches, got {total_matches}')
              sys.exit(1)
          print(f'OK: {total_matches} image matches (>= 10 required)')
          PY

      - name: Validate renovate.json
        run: |
          npx --yes renovate-config-validator renovate.json

  tag:
    needs: lint
    runs-on: ubuntu-24.04
    # B3 Path A: auto-tag patch on every push-to-main that passes lint.
    # Renovate-PR merges land here; their tag triggers my-kluster's Renovate scan
    # which is what closes SC#2 (REQ-pr-to-cluster-latency).
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    permissions:
      contents: write
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
      - name: Auto-tag patch bump
        uses: mathieudutour/github-tag-action@v6.2
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          default_bump: patch
          tag_prefix: v
          release_branches: main
```

- `actions/checkout@v4` consistent with arrconf-image.yml
- `azure/setup-helm@v4` pinned to v3.18.0 (the floor for app-template 5.0.0; helm 3.18 is the minimum compatible version per RESEARCH §"Breaking Changes")
- kubeconform installed via direct curl (no third-party action) — consistent with arrconf-image.yml's "install binaries on the fly" pattern

- `tag` job gated to push-to-main (B3) and requires `contents: write` permission. The action `mathieudutour/github-tag-action@v6.2` is the documented stable; if v6.3+ is available at execute time, bump and capture in SUMMARY.
- The `tag` job runs ONLY after `lint` passes — no half-baked tags.

```json
{
  "$schema": "https://docs.renovatebot.com/renovate-schema.json",
  "extends": ["config:recommended"],
  "customManagers": [
    {
      "customType": "regex",
      "fileMatch": ["^charts/arr-stack/values\\.yaml$"],
      "matchStrings": [
        "#\\s*renovate:\\s*image=(?<depName>[^\\s]+)\\s*\\n\\s*repository:\\s*[^\\s]+\\s*\\n\\s*tag:\\s*\"?(?<currentValue>[^\"\\s]+)\"?"
      ],
      "datasourceTemplate": "docker",
      "versioningTemplate": "docker"
    }
  ],
  "packageRules": [
    {
      "description": "Automerge minor/patch image bumps tracked by customManagers",
      "matchManagers": ["custom.regex"],
      "matchUpdateTypes": ["minor", "patch", "digest"],
      "automerge": true,
      "automergeType": "pr",
      "platformAutomerge": true
    },
    {
      "description": "Automerge minor/patch app-template (Chart.yaml) bumps",
      "matchManagers": ["helmv3"],
      "matchPackageNames": ["app-template"],
      "matchUpdateTypes": ["minor", "patch"],
      "automerge": true,
      "automergeType": "pr",
      "platformAutomerge": true
    },
    {
      "description": "Gate major bumps for manual review",
      "matchUpdateTypes": ["major"],
      "automerge": false
    }
  ]
}
```

The regex captures:
- `depName` from the `# renovate: image=<repo>` line (e.g., `lscr.io/linuxserver/sonarr`)
- `currentValue` from the `tag:` line (e.g., `4.0.17` — handles both quoted and unquoted, optional whitespace, optional intermediate properties)

The regex tolerates exactly 1 line between `repository:` and `tag:` (the values.yaml from Plan 04-03 places them on adjacent lines). If the synthetic test (Task 6.1) returns < 10 matches, the regex needs to be relaxed — but with the current values.yaml shape, 10 image stanzas should each match.
