        # Anonymous query of the GHCR registry: GET /v2/<package>/manifests/<tag>
        # GHCR public packages return 200 without auth.
        STATUS=$(curl -s -o /dev/null -w "%{http_code}" \
          "https://ghcr.io/v2/tom333/arr-stack-arrconf/manifests/0.1.4" \
          -H "Accept: application/vnd.docker.distribution.manifest.v2+json")
        echo "GHCR manifest HTTP status (anon): $STATUS"
        # 200 if public manifest pull succeeds; 401 if private; 404 if tag missing.
        # Note: GHCR may require an anonymous bearer token via GET /token first; if 401, request a token then retry.
        if [ "$STATUS" = "401" ]; then
          TOKEN=$(curl -s "https://ghcr.io/token?service=ghcr.io&scope=repository:tom333/arr-stack-arrconf:pull" | jq -r .token)
          STATUS=$(curl -s -o /dev/null -w "%{http_code}" \
            "https://ghcr.io/v2/tom333/arr-stack-arrconf/manifests/0.1.4" \
            -H "Accept: application/vnd.docker.distribution.manifest.v2+json" \
            -H "Authorization: Bearer $TOKEN")
          echo "GHCR manifest HTTP status (anon-bearer): $STATUS"
        fi
        [ "$STATUS" = "200" ] || { echo "ERROR: GHCR manifest pull failed with status $STATUS"; exit 1; }
  </action>
