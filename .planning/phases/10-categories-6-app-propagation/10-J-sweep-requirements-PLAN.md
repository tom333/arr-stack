from __future__ import annotations

def _strip_metadata(d: dict) -> dict:
    """Drop _caveat/_generated/_source_yaml so byte-equivalence comparison ignores them."""
    return {k: v for k, v in d.items() if not k.startswith("_")}


def _has_action_other_than_no_op(plans: dict) -> tuple[bool, list[str]]:
    """Walk the plan dict; return (drift_present, list of (app, plan_action) pairs that aren't NO_OP)."""
    drift: list[str] = []
    for app, app_plan in plans.items():
        if app.startswith("_"):
            continue
        # Each app's plan is either a list[dict] (sonarr/radarr/prowlarr/qbit) or
        # a {completed, actions_taken} dict (seerr/jellyfin per phase9 fixture shape).
        if isinstance(app_plan, list):
            for entry in app_plan:
                action = entry.get("action") if isinstance(entry, dict) else None
                if action and action != "no-op":
                    drift.append(f"{app}: action={action} name={entry.get('name')}")
        elif isinstance(app_plan, dict):
            taken = app_plan.get("actions_taken")
            if taken:
                drift.append(f"{app}: actions_taken={taken}")
    return (len(drift) > 0, drift)


@pytest.fixture

def test_phase10_sweep_second_run_zero_drift(production_cfg: RootConfig) -> None:
    """SC#2 dispositive: 2nd dry_run against the same fixture state emits 0 plan_action.

    The first run produces the baseline shape (which captures both Categories-
    derived plans for empty manual sections AND override-merge-preserved plans
    for non-empty manual sections — production has BOTH). The second run
    against the SAME respx fixture set must produce NO drift because:
    - Idempotent reconcilers re-emit NO_OP on matched current==desired.
    - FP fixes (10-C qBit / 10-F Seerr user / 10-H Prowlarr) prevent false UPDATEs.
    - dry_run=True means no state mutation between rounds.
    """
    run1 = dry_run_all_apps(production_cfg)
    run2 = dry_run_all_apps(production_cfg)

    drift_present, drift_details = _has_action_other_than_no_op(run2)
    assert not drift_present, (
        f"SC#2 FAILED: 2nd-run drift detected — idempotence broken.\n"
        f"Drift details: {drift_details}\n"
        f"This means one of the 3 FP fixes (qBit #1, Seerr user #3, Prowlarr #2) "
        f"is NOT working as expected, OR a reconciler has a non-idempotent path."
    )

    # Soft check: run1 and run2 should be structurally identical (byte-equivalence).
    # This is stricter than just "run2 has no drift" but cheap to verify.
    assert _strip_metadata(run1) == _strip_metadata(run2), (
        "Run 1 and run 2 produced different plans — likely a non-deterministic "
        "ordering or a timestamp leak in the walker output. Investigate."
    )
