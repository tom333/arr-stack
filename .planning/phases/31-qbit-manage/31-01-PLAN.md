```python
_QBM_HEADER: Final[str] = (
    "# GENERATED by 'arrconf generate' from intent.yml — DO NOT EDIT BY HAND\n"
)


def generate_qbit_manage(cfg: QbitManageConfig) -> str:
    """Pure function: QbitManageConfig → committed config.yml content string.

    Key invariants enforced unconditionally (QBM-02/D-03/D-04):
    - cat_update: false  (never touches qBit categories)
    - cat: {}            (empty — no category management)
    - tag_nohardlinks: true (safe observability tag)
    - rem_orphaned / rem_unregistered: from cfg (default False)
    - recyclebin: enabled: true, empty_after_x_days from cfg

    Security: !ENV tags emitted as YAML literal strings via string construction —
    ruyaml.dump() cannot produce '!ENV VAR_NAME' tagged scalars without a custom
    representer (Pitfall 1 in RESEARCH.md — avoided by string construction).
    Determinism: explicit section/key order in the formatted string.
    """
    lines: list[str] = [_QBM_HEADER]
    lines += [
        "qbt:",
        f"  host: {cfg.qbt_host}",
        "  user: !ENV QBT_USER",
        "  pass: !ENV QBT_PASS",
        "",
    ]
    lines += [
        "settings:",
        "  force_auto_tmm: false",
        "  cat_update: false            # QBM-02: arrconf is sole category owner",
        "  tag_update: true",
        f"  rem_unregistered: {'true' if cfg.rem_unregistered else 'false'}",
        f"  rem_orphaned: {'true' if cfg.rem_orphaned else 'false'}",
        "  tag_nohardlinks: true        # D-03: observability only",
        "  skip_cleanup: false",
        "  dry_run: false",
        "  share_limits: true",
        "  recyclebin:",
        "    enabled: true",
        f"    empty_after_x_days: {cfg.recyclebin_days}",
        "    save_torrents: false",
        "",
    ]
    lines += ["cat: {}", ""]
    lines.append("tracker_tags:")
    for entry in cfg.tracker_tags:
        lines += [f"  {entry.keyword}:", f"    tag: {entry.tag}"]
    lines += ["  other:", "    tag: public", ""]
    lines.append("share_limits:")
    for grp in sorted(cfg.share_limits, key=lambda g: g.priority):
        lines += [
            f"  {grp.name}:",
            f"    priority: {grp.priority}",
            "    include_all_tags:",
            f"      - {grp.tracker_tag}",
            f"    max_ratio: {grp.max_ratio}",
            f"    max_seeding_time: {grp.max_seeding_time}",
            f"    min_seeding_time: {grp.min_seeding_time}",
            f"    cleanup: {'true' if grp.cleanup else 'false'}",
        ]
    # RESEARCH.md A1 — the default group omits include_all_tags, assumed to mean
    # "match all" (catch-all for untagged torrents). UNVERIFIED for qbit_manage
    # v4.6.6: verify post-deploy that untagged torrents actually match this group.
    lines += [
        "  default:",
        "    priority: 999",
        "    max_ratio: 2.0",
        "    max_seeding_time: 10080",
        "    min_seeding_time: 0",
        "    cleanup: false",
        "",
    ]
    lines += ["nohardlinks: {}", ""]
    return "\n".join(lines)
```

def test_generate_qbit_manage_yaml_valid() -> None:
