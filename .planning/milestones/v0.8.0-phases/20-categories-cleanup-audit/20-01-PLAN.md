```python
def _norm_path(p: str) -> str:
    """Strip trailing slash for consistent comparison (Pitfall 7)."""

    """A path is legacy iff it's in LEGACY_PATHS_HARD OR it's not a known Category path."""
    n = _norm_path(p)
```

```python
def _render_table(headers: list[str], rows: list[list[str]]) -> str:
    """Emit a GFM table. Escape pipes in cell content (RESEARCH.md §Security)."""
    def cell(c: Any) -> str:
        s = str(c) if c is not None else ""
        return s.replace("|", "\\|").replace("\n", " ")
    sep = "|" + "|".join(["---"] * len(headers)) + "|"
    head = "|" + "|".join(headers) + "|"
    body = ["|" + "|".join(cell(c) for c in row) + "|" for row in rows]
    return "\n".join([head, sep, *body])
```
