def _strip_metadata(d: dict) -> dict:
    """Drop _caveat/_generated/_source_yaml so byte-equivalence comparison ignores them."""
    return {k: v for k, v in d.items() if not k.startswith("_")}


def _has_action_other_than_no_op(plans: dict) -> tuple[bool, list[str]]:
    """Walk the plan dict; return (drift_present, list of (app, plan_action) pairs that aren't NO_OP)."""
    drift: list[str] = []
    for app, app_plan in plans.items():
        if app.startswith("_"):
            continue
        # Each app's plan is either a list[dict] (sonarr/radarr/prowlarr/qbit) or
        # a {completed, actions_taken} dict (seerr/jellyfin per phase9 fixture shape).
        if isinstance(app_plan, list):
            for entry in app_plan:
                action = entry.get("action") if isinstance(entry, dict) else None
                if action and action != "no-op":
                    drift.append(f"{app}: action={action} name={entry.get('name')}")
        elif isinstance(app_plan, dict):
            taken = app_plan.get("actions_taken")
            if taken:
                drift.append(f"{app}: actions_taken={taken}")
    return (len(drift) > 0, drift)
